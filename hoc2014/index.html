How can I make my sketch more interactive?

If statements


void setup() {
    size(300, 300);
}

void draw() {
    // if the mouse is above the 150th pixel
    if(mouseY < 150) {
        fill(255, 0, 0); 
    }
    // if the mouse is below the 150th pixel
    if(mouseY >= 150) {
        fill(0, 255, 0);
    }
    ellipse(150, 150, 100, 100);
}

Else statement


void setup() {
    size(300, 300);
}

void draw() {
    // if the mouse is above the 150th pixel
    if(mouseY < 150) {
        fill(255, 0, 0);
    // otherwise 
    } else {
        fill(0, 255, 0);
    }
    ellipse(150, 150, 100, 100);
}

For loop


void setup() {
    size(300, 300);
    fill(0, 0, 255);
}

void draw() {
    for (int i = 15; i < 300; i = i + 30) {
        ellipse(i, 150, 30, 30);
    }
}

Nested For loop


void setup() {
    size(300, 300);
    fill(0, 0, 255);
}

void draw() {
    for (int i = 15; i < 300; i = i + 30) {
        for (int j = 15; j < 300; j = j + 30) {

Click with math


void setup() {
  size(512, 512);  
}

void draw() {
  fill(255, 0, 0);
  ellipse(256, 256, 100, 100);  
}

void mousePressed() {
  // a^2 + b^2 < c^2, Pythagorean Theorem
  if (sq(mouseX-256)+sq(mouseY-256) < sq(50)) {
    println("Circle! :)");
  } else {
    println("Not circle! :(");
  }
}
mousePressed, mouseX, and mouseY are all built in to processing

Fractals


void setup() {
    size(512, 512);  
}
void draw() {
    square(0, 0, 512, 512, 255);
}
void square(int x, int y, int w, int h, int c) {
    fill(c);
    rect(x, y, w, h);
    if (w > 1) {
        square(x, y, w/3, h/3, 255 - c);
        square(x + w - w/3, y, w/3, h/3, 255 - c);
        square(x, y + h - h/3, w/3, h/3, 255 - c);
        square(x + w - w/3, y + h - h/3, w/3, h/3, 255 - c);
    }
}
