Explore digital signal


int SPEAKER = P2_0;
int BUTTON = P1_3;
int highDelay = 2;
int lowDelay = 3;
void setup()
{
  pinMode(SPEAKER, OUTPUT);
  pinMode(BUTTON, INPUT_PULLUP);
}
void loop()
{
  if(digitalRead(BUTTON) == LOW) { // keep your friends
    digitalWrite(SPEAKER,HIGH);
    delay(highDelay);
    digitalWrite(SPEAKER,LOW);
    delay(lowDelay);
  }
}
