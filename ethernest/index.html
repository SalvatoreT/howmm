How do basic sketches work?

Two main parts


void setup() 
{
  // called at the beginning
}

void draw() 
{
  // run every refresh
  // default 60 frames per second
}

Set screen size


void setup() 
{
  size(400, 600); // width, height (pixels)
}
size has to be called first

Draw something


void setup() 
{
  size(400, 600);
}

void draw() 
{
  fill(255, 0, 0); // make fill color red
  // centered at x, y and width, height
  ellipse(200, 200, 300, 300);
}
the origin is the top left corner

Draw order


void setup() 
{
  size(400, 600);
}

void draw() 
{
  fill(255, 0, 0);
  ellipse(200, 200, 300, 300);
  fill(0, 255, 0);
  // top left x,y and width, height
  rect(50, 250, 300, 300);
}
please note that the rectangle's location is based on the top left corner, circle the middle

How can I make my sketch more interactive?

If statements


void setup() {
    size(300, 300);
}

void draw() {
    // if the mouse is above the 150th pixel
    if(mouseY < 150) {
        fill(255, 0, 0); 
    }
    // if the mouse is below the 150th pixel
    if(mouseY >= 150) {
        fill(0, 255, 0);
    }
    ellipse(150, 150, 100, 100);
}

Else statement


void setup() {
    size(300, 300);
}

void draw() {
    // if the mouse is above the 150th pixel
    if(mouseY < 150) {
        fill(255, 0, 0);
    // otherwise 
    } else {
        fill(0, 255, 0);
    }
    ellipse(150, 150, 100, 100);
}

How can I click things?

Click


void setup() {
  size(512, 512);  
}

void draw() {  
}

// special method, every mouse click
void mousePressed() {
  println("Mouse pressed at ("+mouseX+","+mouseY+")");
}
mousePressed, mouseX, and mouseY are all built in to processing

Click better


void setup() {
  size(512, 512);  
}

void draw() {
  fill(255, 0, 0);
  ellipse(256, 256, 100, 100);  
}

void mousePressed() {
  // a^2 + b^2 < c^2, Pythagorean Theorem
  if (sq(mouseX-256)+sq(mouseY-256) < sq(50)) {
    println("Circle! :)");
  } else {
    println("Not circle! :(");
  }
}
mousePressed, mouseX, and mouseY are all built in to processing
